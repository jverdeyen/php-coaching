# waarom ?

    mb_strlen() vs. strlen()

# tip
bovenaan php file

    @@@ php
    // Set internal character encoding to UTF-8
    mb_internal_encoding("UTF-8");

    // Display current internal character encoding
    echo mb_internal_encoding();

vlak voor output

    @@@ php
    // Tell PHP that we output UTF-8 to broweser
    mb_http_output("UTF-8");
